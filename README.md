## Example
``` html

<script src="https://unpkg.com/countryblock.js/dist/js/countryblock.min.js"></script>
<script>
countryBlock({
	status:true,
	country:["TR","USA"],
	locationHref:"https://github.com/mestoness/countryblock.js"	
});
</script>

```
